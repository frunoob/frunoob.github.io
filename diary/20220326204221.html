# 分页纯前端实现

<template>
  <div class="touch">
    <div
      v-for="(item, index) in this.totalPages[this.currentPage]"
      :key="index"
    >
      <p class="list">
        <a :href="item.path" ">{{ item.title }} </a>
      </p>
    </div>
    <div class="pageManage">
      <span>第{{ this.currentPage + 1 }}页</span>
      <button id="prePage" @click="prePage">上一页</button>
      <button id="nextPage" @click="nextPage">下一页</button>
      <input
        id="input"
        placeholder="页数"
        type="number"
        v-model="input"
        @keydown.enter="gotopage"
      /><button  @click="gotopage"  id="gotopage">
        确定
      </button>
    </div>
  </div>
</template>

export default {
  data() {
    return {
      data: [],
      totalPages: [], //按照每页指定数量数据，分好组的全部数据
      pageSize: 20,
      pageNum: 1, //共几页
      dataShow: [],
      currentPage: 0,
      input: null,
    };
  },
  methods: {
    // 跳转到指定页面
    gotopage() {
      // console.log(this.input);
      if (0 < this.input && this.input <= this.pageNum) {
        this.currentPage = this.input - 1;
      } else {
        document.getElementById("input").value = this.currentPage + 1;
      }
    },
    //下一页
    nextPage() {
      //   console.log(this.pageNum);
      if (this.currentPage + 1 < this.pageNum) {
        //当前页数等于总页数时就禁用按钮
        this.currentPage++;
        if (this.currentPage > 0) {
          document.getElementById("prePage").disabled = false;
        }
      } else {
        document.getElementById("nextPage").disabled = true;
      }
    },
    //上一页
    prePage() {
      if (this.currentPage > 0) {
        this.currentPage--;
        if (this.currentPage + 1 < this.pageNum) {
          document.getElementById("nextPage").disabled = false;
        }
      } else {
        document.getElementById("prePage").disabled = true;
      }
    },
  },
  created() {
    // 获取页面数组
    this.data = this.$site.pages.filter(
      (item) => item.path.indexOf("/list/") == 0
    ).reverse();
    // 获得总页数
    this.pageNum = Math.ceil(this.data.length / this.pageSize) || 1;
    for (let i = 0; i < this.pageNum; i++) {
      this.totalPages[i] = this.data.slice(
        this.pageSize * i,
        this.pageSize * (i + 1)
      );
    }
    console.log(this.totalPages[this.currentPage]);
  },
};
