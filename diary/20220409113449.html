# 链表内指定区间反转

题目的主要信息：

将一个节点数为 size 链表 m 位置到 n 位置之间的区间反转
链表其他部分不变，返回头节点

思路

先取出要进行反转的子链表并对其进行反转操作，之后再将其接回原来的主链表中即可

ListNode结点及链表定义类

public class ListNode {
    int val;
    ListNode next = null;
    ListNode (){}
    ListNode(int num){
        this.val = num;
    }
    //构造单向链表
    ListNode returnList(int size){
        ListNode head = new ListNode(-1);
        size++;
        while(--(size)>0){
            ListNode temp = new ListNode(size);
            temp.next = head.next;
            head.next=temp;
        }
        return head.next;
    }
    // 打印单向链表
    public static void printList(ListNode list){
        while(list!= null){
            System.out.print(list.val);
            list = list.next;
            if(list!=null){
                System.out.print("-->");
            }
        }
    }
}

具体实现

public class BM2 {
    /**
     * @param head ListNode类
     * @param m    int整型
     * @param n    int整型
     * @return ListNode类
     */
    public ListNode reverseBetween(ListNode head, int m, int n) {
        // 建立一个虚拟的头节点
        ListNode dummyHead = new ListNode(-1);
        dummyHead.next =head;
        // 找到子链的首结点的前一个结点
        ListNode pre = dummyHead;
        for (int i = 1; i < m; i++){
            pre = pre.next;
        }
        // 找到子链的左端结点
        ListNode leftNode = pre.next;
        // 找到子链的最后一个结点
        ListNode rightNode=leftNode;
        for(int i = m; i < n; i++){
            rightNode = rightNode.next;
        }
        // 截取子链  获得左右端结点  并获取子链前一个和后一个结点  方便之后将子链连接回去
        ListNode after  = rightNode.next;
        // 切断连接为后续子链的反转做准备
        pre.next = null;
        rightNode.next = null;
        //子链表反转
        rightNode = reverse(leftNode);
        // 接回原来的子链
        pre.next = rightNode;
        leftNode.next = after;
        //return head;  // 如果子链是从head结点开始的话，那么子链调转之后，head指向的子链的最后一个结点了，就不是总链的首个结点了
        return dummyHead.next;
    }
    public ListNode reverse(ListNode head){
        ListNode pre = null;
        while(head!=null){
            ListNode temp = head.next;
            head.next = pre;
            pre = head;
            head  = temp;
        }
        return pre;
    }
    public static void main(String[] args) {
        BM2 bm2 = new BM2();
        ListNode list = new ListNode().returnList(15);
        ListNode.printList(list);
        System.out.println();
        bm2.reverseBetween(list,2,4);
        ListNode.printList(list);
    }
}
