# 数据库自动分区

# partition 的一些限制

  1. 分区所涉及的列必须是每一个主键和唯一键的子集

  2. 关于InnoDB引擎的限制 InnoDB存储引擎中，分区和外键不兼容，不能同时使用。 ALTER TABLE ... OPTIMIZE PARTITION语句不能正常使用，用ALTER TABLE ... REBUILD PARTITION and ALTER TABLE ... ANALYZE PARTITION代替 不要给任何分区或任何子分区指定存储引擎，因为同一个表的所有分区的存储引擎必须一致。

  3. 关于一些sql函数的限制
    只有以下函数能够在分区表达式中使用（注意不是mysql语句，是创建表时的分区表达式） ABS()

    CEILING() (see CEILING() and FLOOR())

    DATEDIFF()

    DAY()

    DAYOFMONTH()

    DAYOFWEEK()

    DAYOFYEAR()

    EXTRACT() (see EXTRACT() function with WEEK specifier)：有个BUG：指定单元为 WEEK 时，不允许将其作为分区函数

    FLOOR() (see CEILING() and FLOOR())

    HOUR()

    MICROSECOND()

    MINUTE()

    MOD()

    MONTH()

    QUARTER()

    SECOND()

    TIME_TO_SEC()

    TO_DAYS()

    TO_SECONDS()

    UNIX_TIMESTAMP() (with TIMESTAMP columns)

    WEEKDAY()

    YEAR()

    YEARWEEK()

  4. 关于锁的限制 select 只锁包含要查询的数据的分区 update 只有当没有分区列被更新的时候才不用全表锁 insert 只有要被新增的分区加锁。但如果设置了自增列，则全表全分区锁

CREATE TABLE employees (

通过sql语句来创建带有分区选项的表。

创建分区的提示

分区的定义需要从低到高来排序，不可以打乱顺序。

提高查询速度的建议

where查询条件最好包含分区列 并使用=,<, >, <=, >=, and <>或者IN () 表达式，这样查询的时候mysql会自动忽略不包含目标数据的分区，只查询包含目标数据的分区，从而提高查询速度

注意null

在list分区中，插入null数据会导致报错， 必须显式定义一个包含null的list分区

在hash 和range中，null被当作是最小的 存放在第一个分区

# 使用分区遇到的一些问题

数据库表使用了主键，但是由于主键必须包含分区列。此时可以删除原有主键，创建一个新的联合主键，新的主键不能是null；

ALTER TABLE <table>
add primary key(<column1>,<column2>);

alter table <table_name>

下面是创建存储过程的代码

create definer = root@`%` procedure add_partition_if_not_exists(IN t_name varchar(64), IN partition_date date, IN partition_type varchar(10), IN prefix varchar(20))
BEGIN
    DECLARE p_name VARCHAR(30);
    DECLARE partition_exists INT;
    IF partition_type = 'DAILY' THEN
        SET p_name = DATE_FORMAT(partition_date, CONCAT(prefix,'%Y%m%d'));
        SET @partition_value = CONCAT('TO_DAYS(''', DATE_ADD(partition_date,interval 1 day ), ''')');
    ELSEIF partition_type = 'MONTHLY' THEN
        SET p_name = DATE_FORMAT(partition_date, CONCAT(prefix,'%Y%m'));
        SET @partition_value = CONCAT('TO_DAYS(''', DATE_ADD(partition_date,interval 1 month), ''')');
    ELSEIF partition_type = 'YEARLY' THEN
        SET p_name = DATE_FORMAT(partition_date, CONCAT(prefix,'%Y'));
        SET @partition_value = CONCAT('TO_DAYS(''', DATE_ADD(partition_date,interval 1 year), ''')');
    END IF;
    -- 检查分区是否存在
    SET partition_exists = (SELECT COUNT(1)
                            FROM information_schema.PARTITIONS
                            WHERE TABLE_SCHEMA = DATABASE()
                              AND TABLE_NAME = t_name
                              AND PARTITION_NAME = p_name);
    -- 如果分区不存在，则添加
    IF partition_exists = 0 THEN
        SET @sql = CONCAT('ALTER TABLE ', t_name, ' REORGANIZE PARTITION ',prefix,'other INTO ( PARTITION ', p_name, ' VALUES LESS THAN (', @partition_value, '),PARTITION ',prefix,'other VALUES LESS THAN MAXVALUE)');
        PREPARE stmt FROM @sql;
        EXECUTE stmt;
        DEALLOCATE PREPARE stmt;
    END IF;
END;

如何调用

要调用改存储过程，只需要执行call add_partition_if_not_exists(表名，分区日期，分区类型，分区名称前缀prefix) 其中，分区类型有三种分别是 DAILY,MONTHLY, YEARLY，举例说明:DAILY类型代表按日分区，每天分区,分区名称格式是<prefix>20240607，该分区的对应的值是2024年6月8号0点的时间戳;同理，MONTHLY代表按月分区，分区名称格式是<prefix>202406,对应的值是下一个月1号的时间戳;最后一个代表的按年分区，分区名称格式是<prefix>2024，对应的值是下一年的一月一号的时间戳。

可以再创建一个存储过程

create definer = root@`%` procedure daily_manage_partitions()
BEGIN
    DECLARE current_now_d DATE;
    DECLARE current_now_m DATE;
    DECLARE current_now_y DATE;
    DECLARE end_date_d DATE;
    DECLARE end_date_m DATE;
    DECLARE end_date_y DATE;
    DECLARE p_type_day VARCHAR(20);
    DECLARE p_type_month VARCHAR(20);
    DECLARE p_type_year VARCHAR(20);
#    分区类型
    SET p_type_day = concat('DAILY');
    SET p_type_month = concat('MONTHLY');
    SET p_type_year = concat('YEARLY');
#   起始时间
    SET current_now_d = curdate();
    set current_now_m = DATE(DATE_FORMAT(NOW(), '%Y-%m-01'));
    set current_now_y = DATE(DATE_FORMAT(NOW(), '%Y-01-01'));
    #结束时间
    SET end_date_d = DATE_ADD(current_now_d,INTERVAL 1 year );
    set end_date_m = DATE_ADD(current_now_m, interval 1 year );
    set end_date_y = DATE_ADD(current_now_y, interval  2 year );
    # 从今天开始，为'my_table_d'自动创建未来一年的分区（包含本日）,按日进行分区,每天一个分区，分区名称的前缀是 'd_'
    WHILE  current_now_d < end_date_d DO
            -- 需要修改表明和分区名称前缀
            CALL add_partition_if_not_exists('my_table', current_now_d,p_type_day,'d_');
            SET current_now_d = DATE_ADD(current_now_d, interval 1 day );
        END WHILE;
   # 从本月开始，为'my_table_m'自动创建未来一年的分区（包含本月），按月进行分区，每个月一个分区，分区的名称的前缀是 'm_'
    WHILE  current_now_m < end_date_m DO
            -- 需要修改表明和分区名称前缀
            CALL add_partition_if_not_exists('my_table_m', current_now_m,p_type_month,'m_');
            SET current_now_m = DATE_ADD(current_now_m, interval 1 month );
        END WHILE;
   # 从本年开始，为'my_table_y'自动创建两年的分区（包含本年），按年进行分区，没年一个分区，分区的名称的前缀是 'y_'
    WHILE  current_now_y < end_date_y DO
            -- 需要修改表明和分区名称前缀
            CALL add_partition_if_not_exists('my_table_y', current_now_y,p_type_year,'y_'); 
            SET current_now_y = DATE_ADD(current_now_y, interval 1 year );
        END WHILE;
END;

最后只需要开启event_scheduler并创建新的event就可以实现每天自动创建分区了

-- 开启event_scheduler
set global  event_scheduler  = ON;
-- 创建event
create definer = root@`%` event scheduled_task on schedule
        -- 每两个小时执行一个，如果要每天的话就是 1 DAY 每个月（1 MONTH） 每秒(1 second) 每分钟(1 minute)
	every '2' hour
        -- 开始时间，now() 代表是当前时间，从当前时间开始，每两个小时执行一次；DATE_ADD(now(),interval 5 second )代表的是5秒之后执行
	starts DATE_ADD(now(),interval 5 second )
	enable
	do
	BEGIN
        call daily_manage_partitions;
    	END;
-- 查询时间的执行记录
select * from mysql.event;
