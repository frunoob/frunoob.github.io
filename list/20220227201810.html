class CQueue {
  Deque<Integer> arr1,arr2;
    public  CQueue() {
         arr1=  new LinkedList<>();
         arr2=  new LinkedList<>();
    }
    public void appendTail(int value) {
        if (arr1.isEmpty()){
            arr1.push(value);
        }else{
            while (!arr1.isEmpty()){
                arr2.push(arr1.pop());
            }
            arr2.push(value);
            while (!arr2.isEmpty()){
                arr1.push(arr2.pop());
            }
        }
    }
    public int deleteHead() {
        if (arr1.isEmpty()){
            return -1;
        }
        return  arr1.pop();
    }
}
